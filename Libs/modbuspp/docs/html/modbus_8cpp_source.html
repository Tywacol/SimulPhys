123 void modbus::modbus_write(int address, int amount, int func, uint16_t *value) {
124  if(func == WRITE_COIL || func == WRITE_REG) {
125  uint8_t to_send[12];
126  modbus_build_request(to_send, address, func);
127  to_send[5] = 6;
128  to_send[10] = (uint8_t) (value[0] >> 8);
129  to_send[11] = (uint8_t) (value[0] & 0x00FF);
130  modbus_send(to_send, 12);
131  } else if(func == WRITE_REGS){
132  uint8_t to_send[13 + 2 * amount];
133  modbus_build_request(to_send, address, func);
134  to_send[5] = (uint8_t) (5 + 2 * amount);
135  to_send[10] = (uint8_t) (amount >> 8);
136  to_send[11] = (uint8_t) (amount & 0x00FF);
137  to_send[12] = (uint8_t) (2 * amount);
138  for(int i = 0; i < amount; i++) {
139  to_send[13 + 2 * i] = (uint8_t) (value[i] >> 8);
140  to_send[14 + 2 * i] = (uint8_t) (value[i] & 0x00FF);
141  }
142  modbus_send(to_send, 13 + 2 * amount);
143  } else if(func == WRITE_COILS) {
144  uint8_t to_send[14 + (amount -1) / 8 ];
145  modbus_build_request(to_send, address, func);
146  to_send[5] = (uint8_t) (7 + (amount -1) / 8);
147  to_send[10] = (uint8_t) (amount >> 8);
148  to_send[11] = (uint8_t) (amount >> 8);
149  to_send[12] = (uint8_t) ((amount + 7) / 8);
150  for(int i = 0; i < amount; i++) {
151  to_send[13 + (i - 1) / 8] += (uint8_t) (value[i] << (i % 8));
152  }
153  modbus_send(to_send, 14 + (amount - 1) / 8);
154  }
155 }

452  switch(msg[8]){
